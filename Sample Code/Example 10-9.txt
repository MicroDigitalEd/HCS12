#include <mc9s12dp512.h>     
void SerTx(unsigned char);
void main(void)
  {
    SCI0BDH=0x0;			
    SCI0BDL=0x0D;			
    SCI0CR1=0x0;			
    SCI0CR2=0X0C;  
    while(1)
      {
        SerTx('Y');
        SerTx('E');
        SerTx('S');
      }
  }
void SerTx(unsigned char c)
  {  	
    while(!(SCI0SR1 & SCI0SR1_TDRE_MASK));	
    SCI0DRL=c;				
  }
