#include <mc9s12dp512.h>
#ifndef TFLG1_C6F_MASK
  #define TFLG1_C6F_MASK 0x40
#endif
unsigned int Tcount, T1, T2;
void main(void)
  {
   DDRA = 0xFF;	
   DDRB = 0xFF;	
   TSCR1 = 0x80; 
   TSCR2 = 0x00; 
   TIOS = TIOS & ~0x40;  
   TCTL3 = 0x10;  
   TFLG1 = TFLG1 | TFLG1_C6F_MASK;  
   while (!(TFLG1 & TFLG1_C6F_MASK));
   T1 = TC6; 
   TFLG1 = TFLG1 | TFLG1_C6F_MASK;  
   while (!(TFLG1 & TFLG1_C6F_MASK));
   T2 = TC6; 
   Tcount = T2-T1;
   PORTB = Tcount & 0x00FF;  
   PORTA = ((Tcount & 0xFF00) >> 8);  
   for(;;);
  }
