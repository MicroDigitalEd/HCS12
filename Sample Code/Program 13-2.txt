Entry: LDS	#RAMEnd+1		
	BSET	DDRB,%11111111	
	MOVB	#$82,ATD0CTL2	
	JSR   DELAY20uS	
	MOVB	#$08,ATD0CTL3
 	MOVB	#$E3,ATD0CTL4
	MOVB	#$82,ATD0CTL5	
	CLI				
	BRA	$			
AD_ISR:
	LDAA	ATD0DR0L		
	STAA	PORTB			
	JSR	TDELAY		
	BSET	ATD0CTL2,mATD0CTL2_ASCIF
	MOVB	#$82,ATD0CTL5	
	RTI				
	ORG $FFD2
	DC.W AD_ISR
	ORG $FFFE
	DC.W Entry
