#include <mc9s12dp512.h>
void main(void)
  { 
    unsigned char mydata[] = {0x25,0x62,0x3F,0x52};
    unsigned char sum = 0;
    unsigned char x;
    unsigned char chksumbyte;
    DDRA = 0xFF;
    DDRB = 0xFF;				
    for(x=0;x<4;x++)
      {
        PORTA = mydata[x]; 		
        sum = sum + mydata[x];	
        PORTB = sum;   		
      }
    chksumbyte = ~sum + 1;		
    PORTB = chksumbyte; 		
  }
