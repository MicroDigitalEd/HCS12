interrupt (((0x10000-Vrti)/2)-1) void RTI_ISR(void)
{
  COUNT++;					
  if(COUNT==8) 			
  {
    PORTB = PORTB ^ PORTB_BIT4_MASK; 	
    COUNT = 0;      			
  }
  CRGFLG = CRGFLG | CRGFLG_RTIF_MASK; 	
}
