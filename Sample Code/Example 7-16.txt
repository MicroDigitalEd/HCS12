#include <mc9s12dp512.h>
void MSDelay(unsigned int);
void main(void)
  {
    DDRB = 0xFF;
    DDRA = 0xFF;			
    PORTB=0x55;
    PORTA=0x55;
    while(1)
      {
        PORTB=PORTB^0xFF;	
        PORTA=PORTA^0xFF;
        MSDelay(250);
      }
  }
void MSDelay(unsigned int itime)
  {
    unsigned int i; 
    unsigned int j;
    for(i=0;i<itime;i++)
      for(j=0;j<331;j++);
  }
