#include <mc9s12dp512.h>     
#define MYSW PORTB_BIT5		
void main(void)
{
  unsigned char z;
  unsigned char fname[]="ALI";
  unsigned char lname[]="SMITH";
  DDRB_BIT5 = 0; 			
  SCI0BDH=0x0;			
  SCI0BDL=0x0D;			
  SCI0CR1=0x0;			
  SCI0CR2=0x0C;  
  if(MYSW==0)				
  {
    for(z=0;z<3;z++)		
    {
      while(!(SCI0SR1 & SCI0SR1_TDRE_MASK));	
	 SCI0DRL=fname[z];	
    }
  }
  else
  {
    for(z=0;z<5;z++)			
    {
      while(!(SCI0SR1 & SCI0SR1_TDRE_MASK));
      SCI0DRL=lname[z];	
    }
  }
  while(1);
}
