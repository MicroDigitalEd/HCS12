#include <mc9s12dp512.h>     
#define SW PORTA_BIT7
#define MTR  PORTB_BIT0
void MSDelay(unsigned int value);
void main()
  {
  DDRA=0x7F;       
  DDRB=0x01;       
  while(1)
    {
      if(SW == 1)
        {
          MTR = 0;
          MSDelay(25);
          MTR = 1;
          MSDelay(75);
        }
      else
        {
          MTR = 1;
          MSDelay(50);
          MTR = 0;
          MSDelay(50);
        }
    }
  }
void MSDelay(unsigned int value)
  {
    unsigned char x, y;
    for(x=0; x<1275; x++)
      for(y=0; y<value; y++);
  }
