	BSET DDRB,%00000001		
	BSET DDRB,%00000100		
	BSET DDRB,%00001000		
	BCLR DDRA,%10000000		
MONITOR:
	BRCLR PORTA,%10000000,CLKWISE
	BSET  PORTB,%00000001			
	BCLR  PORTB,%00000010			
	BCLR  PORTB,%00000100			
	BSET  PORTB,%00001000			
	BRA   MONITOR
CLKWISE:
	BCLR PORTB,%00000001			
	BSET PORTB,%00000010			
	BSET PORTB,%00000100			
	BCLR PORTB,%00001000			
	BRA  MONITOR
