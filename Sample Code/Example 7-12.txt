#include <mc9s12dp512.h>
void MSDelay(unsigned int);
void main(void)
  { 
    DDRA = 0xFF;	
    DDRB = 0xFF;	
    DDRK = 0xFF;
    while(1)			
      { 
        PORTA = 0x55;
        PORTB = 0x55;
        PORTK = 0x55;
        MSDelay(250);		
        PORTA = 0xAA;
        PORTB = 0xAA;
        PORTK = 0xAA;
        MSDelay(250);
      }
  }
void MSDelay(unsigned int itime)
  {
    unsigned int i;
    unsigned int j;
    for(i=0;i<itime;i++)
      for(j=0;j<331;j++);
  }
