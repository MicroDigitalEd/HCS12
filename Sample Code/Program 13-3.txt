Entry:
	LDS	#RAMEnd+1		
	BSET	DDRB,%11111111	
	MOVB	#$80,ATD0CTL2	
	JSR	DELAY20uS		
	MOVB	#$20,ATD0CTL3	
	MOVB	#$E3,ATD0CTL4
H1	MOVB	#$82,ATD0CTL5	
H2	BRCLR ATD0STAT0,mATD0STAT0_SCF,H2 
	LDD	ATD0DR0		
	ADDD	ATD0DR1		
	ADDD	ATD0DR2		
	ADDD	ATD0DR3		
	LDX	#4			
	IDIV				
	STX	PORTAB		
	JSR	TDELAY		
	BRA	H1			
DELAY20uS: 
	LDAB #40
D1	DECB
	BNE D1
	RTS
TDELAY: 
	LDD #$0000
D2	DECB
	BNE D2
	DECA
	BNE D2
	RTS
